```python
    """
    從 TaiwanStockInfo 取得發行總股本（SharesIssued）
    單位：元（FinMind 回傳通常為千元，需確認後統一轉換為元）

            return float(df[col].iloc[0])
```

```python
def ols_slope(y: np.ndarray) -> float:
    """計算 OLS 迴歸斜率（最小資料點 3）"""
    if len(y) < 3:
        return np.nan
    x = np.arange(len(y))
    slope, _ = np.polyfit(x, y, 1)
    return slope

def cagr(current: float, past: float, years: float) -> float:
    """計算複合成長率，分母為0或負時回傳 NaN"""
    if pd.isna(current) or pd.isna(past) or past <= 0 or current <= 0:
        return np.nan
    return (current / past) ** (1.0 / years) - 1

def coefficient_of_variation(values: np.ndarray) -> float:
    """計算變異係數 CV = σ/μ，μ=0 時回傳 NaN"""
    mean = np.nanmean(values)
    std = np.nanstd(values)
    if mean == 0 or pd.isna(mean):
        return np.nan
    return std / mean

def winsorize_ratio(values: np.ndarray, col: str = "") -> np.ndarray:
    """比率類指標的極端值壓縮（如法人股本比超過 ±10% 視為極端）"""
    import numpy as np
    return np.clip(values, -0.10, 0.10)  # ±10% 壓縮
```

### 2.5 `core/scorer.py` — 核心重寫（24 子項）

#### 新評分函式模式

不再使用 `five_level_score()` 泛用函式 + thresholds dict，改為每個子項獨立實作階梯邏輯：

```python
def score_trend_structure(row) -> dict:
    """短線-趨勢結構（權重20%）
    
    100分: N=4 且 MA5 > MA10 > MA20 > MA60（多頭排列）
    80分:  N=3
    60分:  N=2
    40分:  N=1
    0分:   N=0 且空頭排列
    """
    n = row.get("N_Trend", 0)
    is_bullish = row.get("Bullish_Alignment", False)
    
    if pd.isna(n):
        return {"score": 0, "details": {"n_trend": None, "note": "資料不足"}}
    
    n = int(n)
    if n == 4 and is_bullish:
        score = 100
    elif n == 3:
        score = 80
    elif n == 2:
        score = 60
    elif n == 1:
        score = 40
    else:
        score = 0
    
    return {"score": score, "details": {"n_trend": n, "is_bullish": bool(is_bullish)}}
```

#### 🆕 權重等比例再分配（金融股專用）

現有系統處理金融股的方式是：跳過特定子項並**給予基準滿分 100**（如 `score_financial_safety_value()` 中回傳 100 分）。

新規格改為：跳過特定子項後，將其權重**等比例分配給其餘子項**，再正常計算加權總分。

**兩者差異示意（價值維度）**：

```
現有系統：
  估值安全(15%)×score + 獲利品質(20%)×score + 成長能力(30%)×score 
  + 財務安全(15%)×100 + 現金流品質(10%)×100 + 股東回報(10%)×score
  = 總分（但 25% 權重被固定為 100 分，造成分數膨脹）

新規格：
  跳過財務安全(15%) + 現金流品質(10%) = 25%
  W_adj = 100% / (100% - 25%) = 1.3333
  估值安全 15%×1.3333 = 20%
  獲利品質 20%×1.3333 = 26.7%
  成長能力 30%×1.3333 = 40%
  股東回報 10%×1.3333 = 13.3%
  總和 = 100% ✅
```

```python
def redistribute_weights(skip_items: list, original_weights: dict) -> dict:
    """
    金融股跳過特定項目時，權重等比例再分配
    
    公式：
    W_adj = 100% / (100% - Σ(skip_weights))
    
    Example:
    價值維度跳過 financial_safety(15%) + cash_flow_quality(10%) = 25%
    W_adj = 100% / (100% - 25%) = 1.3333
    其餘 4 個子項權重各乘以 1.3333
    """
    total_skip = sum(original_weights.get(item, 0) for item in skip_items)
    if total_skip >= 1.0:
        return original_weights  # 防呆，不應發生
    
    ratio = 1.0 / (1.0 - total_skip)
    new_weights = {k: v * ratio for k, v in original_weights.items() if k not in skip_items}
    return new_weights
```

#### 🏗️ 營建股特殊處理（無權重變動）

| 維度 | 子項 | 處理方式 | 說明 |
|------|------|---------|------|
| 波段 | 營收動能 | **資料替代**：營收 YoY → 合約負債 YoY | 因營建股營收認列方式特殊，改用合約負債反映真實業績 |
| 價值 | 財務安全 | **門檻放寬**：Debt_Ratio 40% → 80% | 營建業高負債為行業特性，不應視為風險 |
| 定存 | 無 | 不影響 | 營建股在定存維度與一般股相同處理 |

> 注意：合約負債 YoY 需要在 `processor.py` 中新增計算（從 TaiwanStockBalanceSheet 的 ContractLiabilities type 提取），
> 且在 `score_revenue_momentum()` 中根據 `is_construction()` 判斷是否切換資料源。

## 四、NaN/0 防禦處理標準

所有除法操作前必須進行防禦，統一模式：

```python
# 模式 A：分母可能為 0
result = numerator / denominator.replace(0, np.nan)

# 模式 B：後續處理
result = result.fillna(0).clip(lower=-999, upper=999)  # 防止 Inf

# 模式 C：比率的極端值壓縮
result = result.clip(lower=-10.0, upper=10.0)  # 限制 ±1000%
```

### 檢查清單
- [ ] PE_TTM = close / EPS（EPS≤0 時設為 NaN）
- [ ] ROE_TTM = NI / Equity（Equity=0 時防呆）
- [ ] VR = Vol_MA_5 / Vol_MA_20（Vol_MA_20=0 時防呆）
- [ ] 爆量比 = volume / Vol_MA_5（Vol_MA_5=0 時防呆）
- [ ] 法人股本比% = 累計買超 / SharesIssued（SharesIssued=0 時防呆）
- [ ] CAGR：分母營收≤0 時跳過
- [ ] CV_EPS：μ=0 時回傳 NaN
- [ ] 現金轉換率 = OCF / NI（NI=0 時防呆）
- [ ] 利息保障倍數 = EBIT / Interest（Interest=0 時防呆）

| 價值-財務安全 100分門檻 | Debt_Ratio ≤ 40% **且** Current_Ratio ≥ 200% | Debt_Ratio ≤ **80%** **且** Current_Ratio ≥ 200% | 僅 Debt_Ratio 門檻放寬，權重 15% 不變 |
| 定存維度 | 無特殊處理 | 無特殊處理 | 與一般股完全相同 |

### 🏦 金融股：跳過特定子項 + 權重等比例再分配

| 項目 | 一般股 | 金融股 | 影響 |
|-----|-------|-------|------|
| 短線/波段 | 無特殊處理 | 無特殊處理 | 與一般股完全相同 |
| 價值-財務安全 | 正常評分（權重 15%） | **跳過**，權重分給其餘 4 子項 | 權重變動 ✅ |
| 價值-現金流品質 | 正常評分（權重 10%） | **跳過**，權重分給其餘 4 子項 | 權重變動 ✅ |
| 定存-現金流 | 正常評分（權重 20%） | **跳過**，權重分給其餘 4 子項 | 權重變動 ✅ |
| 定存-財務安全 | 正常評分（權重 15%） | **跳過**，權重分給其餘 4 子項 | 權重變動 ✅ |

```
現有（scorer_v1）：
  score_financial_safety_value()  → 直接回傳 100（分數膨脹）
```

## 八、未來擴充：三種投資風格權重配置

> ⚠️ 此為規劃中的功能，尚未實作。
> 預計在 v2.0 評分邏輯升級的同時或之後導入。

### 設計概念

現有系統只有「一組權重」對應四種維度（短線/波段/價值/定存）。
未來希望提供 **三種投資風格**，讓使用者切換，每種風格對 24 個子項有不同的權重配置：

| 風格 | 定位 | 權重特徵 |
|------|------|---------|
| 🛡️ **保守型** | 低波動、高現金流、高配息 | 價值/定存維度權重 ↑，短線/波段 ↓ |
| ⚖️ **均衡型** | 與現有系統一致 | 等於現有 `SHORT_TERM_WEIGHTS` 等配置 |
| 🚀 **積極型** | 高成長、高動能、高週轉 | 短線/波段維度權重 ↑，價值/定存 ↓ |

### 與 v2.0 升級的關係

- 評分邏輯（24 子項的階梯函式）**不受風格影響**，每個子項的 0-100 分計算是固定的
- 風格只影響 **加權計算**：`∑(子項得分 × 該風格權重)`
- 因此 v2.0 的 24 子項評分重寫 + 風格權重配置是 **互不衝突** 的兩個獨立模組
- 建議合併實施，避免來回改兩次
